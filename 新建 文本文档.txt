// 原有变量 (保留)
const currentPage = ref(1)
const pageSize = ref(10)
const totalRecords = ref(0) // 添加总记录数
const dialogVisible = ref(false)
const searchKeyword = ref('') // 统一的搜索关键词
const isEditing = ref(false) // 是否处于编辑模式

// --- 分类数据 ---
const parentcategorylist = ref<any[]>([])
const soncategorylist = ref<any[]>([])

// --- 表格数据与选中 ---
const activeTab = ref('parent') // 控制显示父类还是子类
const parentTableData = ref<any[]>([])
const childTableData = ref<any[]>([])
const multipleSelection = ref<any[]>([]) // 存储选中的行

// --- 新增/编辑表单 ---
const categoryForm = reactive({
  id: '',
  categoryName: '',
  parentId: '',
  status: '启用',
})

// 原有生命周期和函数 (保留)
onMounted(async () => {
  // 初始化加载数据
  await loadCategoryData()
})

const router = useRouter()

function toProduct() {
  router.push({ name: 'product' })
}

function toCategroy() {
  router.push({ name: 'categroy' })
}

// 加载分类数据
async function loadCategoryData() {
  if (activeTab.value === 'parent') {
    await fetchParentCategories()
  } else {
    await fetchSonCategories()
  }
}

// 获取父级分类
async function fetchParentCategories() {
  try {
    console.log('正在请求父级分类数据...');
    const res = await api.getparentcategory(searchKeyword.value)
    console.log('API响应:', res);
    
    if (res.code === 1) {
      parentcategorylist.value = res.data
      parentTableData.value = res.data.map((item: any) => ({
        ...item,
        status: item.status || '启用', // 默认状态
        parentCategoryName: item.parentCategoryName || '无' // 添加上级分类名称
      }))
      totalRecords.value = res.data.length
      ElMessage.success('父级分类数据加载成功');
    } else {
      ElMessage.error(res.msg || '获取父级分类数据失败')
    }
  } catch (error: any) {
    console.error('获取父级分类数据失败:', error);
    ElMessage.error(error.message || '获取父级分类数据失败')
  }
}

// 获取子级分类
async function fetchSonCategories() {
  try {
    console.log('正在请求子级分类数据...');
    const res = await api.getsoncategory()
    console.log('API响应:', res);
    
    if (res.code === 1) {
      soncategorylist.value = res.data
      childTableData.value = res.data.map((item: any) => ({
        ...item,
        status: item.status || '启用',
        parentCategoryName: item.parentCategoryName || '无'
      }))
      totalRecords.value = res.data.length
      ElMessage.success('子级分类数据加载成功');
    } else {
      ElMessage.error(res.msg || '获取子级分类数据失败')
    }
  } catch (error: any) {
    console.error('获取子级分类数据失败:', error);
    ElMessage.error(error.message || '获取子级分类数据失败')
  }
}

// --- 交互逻辑 ---
// 1. 搜索逻辑：点击搜索按钮时触发 API 请求
async function handleSearch() {
  await loadCategoryData()
}

// 2. 重置逻辑
function resetForm() {
  searchKeyword.value = ''
  multipleSelection.value = []
  loadCategoryData() // 重置后重新加载当前Tab的数据
}

// 3. 切换Tab逻辑
async function switchTab(tab: string) {
  activeTab.value = tab
  multipleSelection.value = []
  await loadCategoryData()
}

// 4. 批量删除逻辑
async function handleBatchDelete() {
  if (multipleSelection.value.length === 0) {
    ElMessage.warning('请先选择要删除的分类')
    return
  }

  try {
    await ElMessageBox.confirm(
      `确定要删除选中的 ${multipleSelection.value.length} 个分类吗?`,
      '警告',
      {
        confirmButtonText: '确定',
        cancelButtonText: '取消',
        type: 'warning',
      }
    )

    // 发送 DELETE 请求删除选中的分类
    for (let i = 0; i < multipleSelection.value.length; i++) {
      const item = multipleSelection.value[i]
      const res = await api.deletecategory(item.id)
      if (res.code !== 1) {
        ElMessage.error(`删除分类 "${item.categoryName}" 失败: ${res.msg}`)
        return
      }
    }
    
    ElMessage.success('删除成功')
    // 重新加载当前 Tab 的数据
    await loadCategoryData()
    multipleSelection.value = [] // 清空选中状态
  } catch (error: any) {
    if (error !== 'cancel') {
      ElMessage.error('删除操作失败')
    }
  }
}

// 5. 删除单个分类
async function handleDelete(row: any) {
  try {
    await ElMessageBox.confirm(
      `确定要删除分类 "${row.categoryName}" 吗?`,
      '警告',
      {
        confirmButtonText: '确定',
        cancelButtonText: '取消',
        type: 'warning',
      }
    )

    const res = await api.deletecategory(row.id)
    if (res.code === 1) {
      ElMessage.success('删除成功')
      // 重新加载当前 Tab 的数据
      await loadCategoryData()
    } else {
      ElMessage.error(res.msg)
    }
  } catch (error: any) {
    if (error !== 'cancel') {
      ElMessage.error('删除操作失败')
    }
  }
}

// 6. 表格多选处理
function handleSelectionChange(val: any[]) {
  multipleSelection.value = val
}

// 7. 状态切换 (禁用/启用)
async function toggleStatus(row: any) {
  // 获取当前状态并切换
  const currentStatus = row.status
  const newStatus = currentStatus === '启用' ? '禁用' : '启用'
  
  try {
    // 构建更新数据对象
    const updateData = {
      id: row.id,
      categoryName: row.categoryName,
      parentId: row.parentId,
      status: newStatus,
      createTime: row.createTime,
      parentCategoryName: row.parentCategoryName
    }
    
    // 发送更新请求
    const res = await api.editcategory(updateData)
    if (res.code !== 1) {
      throw new Error(res.msg)
    }
    
    // 成功后更新本地数据
    row.status = newStatus
    ElMessage.success(`分类 "${row.categoryName}" 状态已更新为 ${newStatus}`)
  } catch (error: any) {
    // 失败时恢复原来的状态
    row.status = currentStatus
    ElMessage.error(error.message || '状态更新失败')
  }
}

// 8. 编辑分类
async function handleEdit(row: any) {
  isEditing.value = true
  dialogVisible.value = true
  // 将行数据复制到表单
  Object.assign(categoryForm, {
    id: row.id,
    categoryName: row.categoryName,
    parentId: row.parentId || '',
    status: row.status || '启用'
  })
}

// 9. 打开新增对话框
function openAddDialog() {
  isEditing.value = false
  dialogVisible.value = true
  // 清空表单
  Object.assign(categoryForm, {
    id: '',
    categoryName: '',
    parentId: activeTab.value === 'child' ? '' : undefined, // 子类需要选择父类，父类不需要
    status: '启用'
  })
}

// 10. 保存分类（新增或编辑）
async function saveCategory() {
  try {
    let res
    if (isEditing.value) {
      // 编辑现有分类
      console.log('正在编辑分类:', categoryForm);
      res = await api.editcategory(categoryForm)
    } else {
      // 新增分类
      console.log('正在添加分类:', categoryForm);
      res = await api.addcategory(categoryForm)
    }
    
    if (res.code === 1) {
      ElMessage.success(isEditing.value ? '编辑成功' : '添加成功')
      dialogVisible.value = false
      // 清空表单
      Object.assign(categoryForm, {
        id: '',
        categoryName: '',
        parentId: '',
        status: '启用'
      })
      // 重新加载数据
      await loadCategoryData()
    } else {
      ElMessage.error(res.msg)
    }
  } catch (error: any) {
    console.error('保存分类失败:', error);
    ElMessage.error(error.message || '操作失败')
  }
}

// --- 表格行样式控制 ---
// 当状态为"禁用"时，给行添加特殊类名
function tableRowClassName({ row }: { row: any }) {
  if (row.status === '禁用') {
    return 'disabled-row'
  }
  return ''
}
</script>
